### Contexto

**Automação: Quando artigo é aprovado na curadoria → criar registro público:**

```sql
CREATE OR REPLACE FUNCTION auto_publish_approved_article()
RETURNS trigger LANGUAGE plpgsql SECURITY DEFINER AS $$
DECLARE
  v_item board_items%ROWTYPE;
  v_authors text[];
  v_author_ids uuid[];
  v_tribe_id int;
BEGIN
  -- Só dispara quando curation_status muda para 'approved'
  IF NEW.action = 'curation_approved' THEN
    SELECT * INTO v_item FROM board_items WHERE id = NEW.item_id;
    
    -- Buscar autores do item (junction table)
    SELECT array_agg(m.name), array_agg(m.id) INTO v_authors, v_author_ids
    FROM board_item_assignments bia
    JOIN members m ON m.id = bia.member_id
    WHERE bia.item_id = NEW.item_id AND bia.role IN ('author', 'contributor');
    
    -- Buscar tribe_id do board
    SELECT pb.tribe_id INTO v_tribe_id
    FROM project_boards pb WHERE pb.id = v_item.board_id;
    
    INSERT INTO public_publications (
      title, abstract, authors, author_member_ids, publication_type,
      tribe_id, cycle_code, board_item_id, is_published
    ) VALUES (
      v_item.title,
      v_item.description,
      COALESCE(v_authors, ARRAY[v_item.title]),
      v_author_ids,
      'article',
      v_tribe_id,
      'cycle3-2026',
      v_item.id,
      false  -- GP/curador publica manualmente após revisão final
    ) ON CONFLICT DO NOTHING;
  END IF;
  
  RETURN NEW;
END; $$;

CREATE TRIGGER trg_auto_publish_approved
  AFTER INSERT ON board_lifecycle_events
  FOR EACH ROW
  WHEN (NEW.action = 'curation_approved')
  EXECUTE FUNCTION auto_publish_approved_article();
```

## W115 — Dashboard por Capítulo (Sponsor + Liaison View)

### Contexto
5 presidentes de capítulo (sponsors) e seus liaisons precisam ver a contribuição
do seu capítulo. Sem isso, renovação de parceria fica baseada em feeling.

### Schema

Não precisa de tabela nova. Todos os dados existem — falta view filtrada.

**RPC: `exec_chapter_dashboard(p_chapter text)`**

```sql
CREATE OR REPLACE FUNCTION exec_chapter_dashboard(p_chapter text)
RETURNS jsonb LANGUAGE plpgsql SECURITY DEFINER AS $$
DECLARE
  v_role text;
  v_is_admin boolean;
  v_desigs text[];
  v_chapter text;
  v_result jsonb;
  v_year_start date;
BEGIN
  -- ACL: admin, sponsor, chapter_liaison, ou membro do capítulo
  SELECT operational_role, is_superadmin, designations, chapter
  INTO v_role, v_is_admin, v_desigs, v_chapter
  FROM members WHERE auth_id = auth.uid();
  
  IF NOT (
    v_is_admin 
    OR v_role IN ('manager', 'deputy_manager')
    OR v_desigs && ARRAY['sponsor', 'chapter_liaison']
    OR v_chapter = p_chapter
  ) THEN
    RETURN jsonb_build_object('error', 'permission_denied');
  END IF;
  
  -- Âncora temporal (kickoff do ano)
  SELECT date INTO v_year_start FROM events
  WHERE type = 'general' AND title ILIKE '%kick%off%'
  AND EXTRACT(year FROM date) = EXTRACT(year FROM now())
  ORDER BY date ASC LIMIT 1;
  v_year_start := COALESCE(v_year_start, make_date(EXTRACT(year FROM now())::int, 1, 1));
  
  SELECT jsonb_build_object(
    'chapter', p_chapter,
    
    -- Membros
    'members', (SELECT jsonb_build_object(
      'total', count(*),
      'active', count(*) FILTER (WHERE current_cycle_active),
      'by_role', (SELECT jsonb_object_agg(operational_role, cnt) 
        FROM (SELECT operational_role, count(*) cnt 
              FROM members WHERE chapter = p_chapter AND current_cycle_active 
              GROUP BY operational_role) sub),
      'tribes', (SELECT jsonb_agg(DISTINCT t.name) 
        FROM members m JOIN tribes t ON t.id = m.tribe_id 
        WHERE m.chapter = p_chapter AND m.current_cycle_active)
    ) FROM members WHERE chapter = p_chapter),
    
    -- Produção
    'production', (SELECT jsonb_build_object(
      'articles_in_pipeline', count(*) FILTER (WHERE bi.curation_status IS NOT NULL),
      'articles_published', count(*) FILTER (WHERE bi.curation_status = 'approved'),
      'board_items_total', count(*)
    ) FROM board_item_assignments bia
    JOIN members m ON m.id = bia.member_id
    JOIN board_items bi ON bi.id = bia.item_id
    WHERE m.chapter = p_chapter AND bi.created_at >= v_year_start),
    
    -- Engajamento
    'engagement', (SELECT jsonb_build_object(
      'attendance_events', count(DISTINCT a.event_id),
      'total_hours', COALESCE(SUM(e.duration_actual / 60.0), 0),
      'avg_attendance_rate', CASE WHEN count(DISTINCT e.id) > 0 
        THEN round(count(DISTINCT a.id)::numeric / count(DISTINCT e.id), 2) 
        ELSE 0 END
    ) FROM attendance a
    JOIN events e ON e.id = a.event_id
    JOIN members m ON m.id = a.member_id
    WHERE m.chapter = p_chapter AND e.date >= v_year_start AND a.status = 'present'),
    
    -- Certificação
    'certification', (SELECT jsonb_build_object(
      'trail_completion_avg', round(AVG(
        COALESCE((SELECT count(*) FROM course_progress cp 
          WHERE cp.member_id = m.id AND cp.status = 'completed'), 0)::numeric 
        / GREATEST((SELECT count(*) FROM courses WHERE is_active), 1) * 100
      ), 1),
      'cpmai_certified', count(*) FILTER (WHERE m.cpmai_certified AND m.cpmai_certified_at >= v_year_start)
    ) FROM members m WHERE m.chapter = p_chapter AND m.current_cycle_active
    AND (m.operational_role IN ('researcher','tribe_leader','manager','deputy_manager','communicator','facilitator')
         OR m.designations && ARRAY['ambassador','curator','co_gp']))
    
  ) INTO v_result;
  
  RETURN v_result;
END; $$;

GRANT EXECUTE ON FUNCTION exec_chapter_dashboard(text) TO authenticated;
```

## W116 — Sistema de Notificações

### Contexto
CXO assessment: "Sem notificações, cada handoff tem 7 dias de atraso."
66 membros, 8 tribos, múltiplos boards — impossível acompanhar tudo via WhatsApp.

### Decisão arquitetural

**Dois canais:**
1. **In-app** (badge no avatar + página de notificações) — real-time
2. **Email digest semanal** — resumo para quem não abre a plataforma todo dia

**NÃO usar push notifications** — requer service worker, permissão do browser, 
complexidade alta para valor marginal neste estágio.

### Schema

```sql
CREATE TABLE IF NOT EXISTS notifications (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  recipient_id uuid REFERENCES members(id) NOT NULL,
  type text NOT NULL,
  title text NOT NULL,
  body text,
  link text,                    -- URL para onde a notificação aponta
  source_type text,             -- 'board_item', 'curation', 'attendance', 'system'
  source_id uuid,               -- ID do item que gerou a notificação
  is_read boolean DEFAULT false,
  read_at timestamptz,
  created_at timestamptz DEFAULT now()
);

CREATE INDEX idx_notif_recipient ON notifications(recipient_id);
CREATE INDEX idx_notif_unread ON notifications(recipient_id, is_read) WHERE is_read = false;
CREATE INDEX idx_notif_created ON notifications(created_at DESC);

-- RLS: membro vê apenas suas notificações
ALTER TABLE notifications ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Members see own notifications" ON notifications
  FOR SELECT TO authenticated
  USING (recipient_id = (SELECT id FROM members WHERE auth_id = auth.uid()));

CREATE POLICY "System can create notifications" ON notifications
  FOR INSERT TO authenticated
  USING (true);

-- Preferências de notificação
CREATE TABLE IF NOT EXISTS notification_preferences (
  member_id uuid PRIMARY KEY REFERENCES members(id),
  in_app boolean DEFAULT true,
  email_digest boolean DEFAULT true,
  digest_frequency text DEFAULT 'weekly',  -- 'daily', 'weekly', 'never'
  muted_types text[] DEFAULT '{}',          -- tipos de notificação silenciados
  updated_at timestamptz DEFAULT now()
);

COMMENT ON TABLE notifications IS 'Notificações in-app para membros';
COMMENT ON TABLE notification_preferences IS 'Preferências de notificação por membro';
```

### Email Digest (Edge Function)

**Edge Function `send-notification-digest`:**
- Cron: semanal (segunda-feira 8h BRT)
- Para cada membro com email_digest=true:
  - Busca notificações da última semana
  - Se count > 0: envia email resumo via Supabase Auth email
  - Template: lista de notificações agrupadas por tipo
  - Assunto: "Núcleo IA & GP — Resumo da semana: X novidades"

**Nota sobre email:** Supabase tem limite de emails no plano free. Para 66 membros
semanais, são ~66 emails/semana = ~10/dia, dentro do limite.
Se exceder, usar Edge Function + Resend (free tier: 100 emails/dia).
